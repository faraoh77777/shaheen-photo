<script>
const CAT_COLORS = {
  dcs:"#F59E0B",lcp:"#F97316",jb:"#EAB308",fi:"#3B82F6",duct:"#06B6D4",
  tray:"#8B5CF6",conduit:"#EC4899",airpipe:"#10B981",hookup:"#14B8A6",
  stanchion:"#6366F1",support:"#A855F7",conn:"#F43F5E",linecheck:"#22C55E",
  looptest:"#0EA5E9",calib:"#D97706",steamtube:"#EF4444",steamins:"#DC2626",
  scaffold:"#64748B",etc:"#6B7280"
};

let loadedImage = null;

// 날짜/시간 기본값
const now = new Date();
document.getElementById("inDate").value = now.toISOString().slice(0,10);
document.getElementById("inTime").value = now.toTimeString().slice(0,5);

function updateCatBar() {
  const cat = document.getElementById("inCat").value;
  document.getElementById("catBar").style.background = CAT_COLORS[cat] || "#6B7280";
}
updateCatBar();

function loadImage(e) {
  const file = e.target.files[0];
  if (!file) return;
  const reader = new FileReader();
  reader.onload = ev => {
    const img = new Image();
    img.onload = () => {
      loadedImage = img;
      // 캠박스에 미리보기
      const camBox = document.getElementById("camBox");
      camBox.innerHTML = `<img src="${ev.target.result}"/>`;
      renderCanvas();
      document.getElementById("previewCard").style.display = "block";
      document.getElementById("previewCard").scrollIntoView({behavior:"smooth"});
    };
    img.src = ev.target.result;
  };
  reader.readAsDataURL(file);
  e.target.value = "";
}

function renderCanvas() {
  if (!loadedImage) return;
  const canvas = document.getElementById("canvas");
  const W = loadedImage.width;
  const H = loadedImage.height;
  canvas.width  = W;
  canvas.height = H;
  const ctx = canvas.getContext("2d");
  ctx.drawImage(loadedImage, 0, 0);

  // 정보 수집
  const date  = document.getElementById("inDate").value;
  const time  = document.getElementById("inTime").value;
  const tag   = document.getElementById("inTag").value.trim();
  const catEl = document.getElementById("inCat");
  const cat   = catEl.options[catEl.selectedIndex].text;
  const loc   = document.getElementById("inLoc").value;
  const note  = document.getElementById("inNote").value.trim();

  const lines = [];
  if (date || time) lines.push(`📅 ${date}  ${time}`);
  if (tag)          lines.push(`🏷️ ${tag}`);
  lines.push(`🔧 ${cat}`);
  lines.push(`📍 ${loc}`);
  if (note)         lines.push(`📝 ${note}`);

  // 폰트 크기 (이미지 너비 기준)
  const fs  = Math.max(18, Math.round(W * 0.030));
  const lh  = Math.round(fs * 1.7);
  const pad = Math.round(fs * 0.7);
  const boxH = lines.length * lh + pad * 2;
  const boxY = H - boxH;

  // 반투명 배경 박스
  ctx.fillStyle = "rgba(6, 13, 26, 0.82)";
  ctx.fillRect(0, boxY, W, boxH);

  // 좌측 컬러 바
  const cat_id = document.getElementById("inCat").value;
  ctx.fillStyle = CAT_COLORS[cat_id] || "#6B7280";
  ctx.fillRect(0, boxY, Math.max(6, Math.round(W * 0.006)), boxH);

  // 텍스트
  ctx.font      = `bold ${fs}px 'Segoe UI', sans-serif`;
  ctx.fillStyle = "#F1F5F9";
  ctx.textBaseline = "top";
  lines.forEach((line, i) => {
    // 첫 줄(날짜) 노란색
    ctx.fillStyle = i === 0 ? "#FFD600" : "#F1F5F9";
    ctx.fillText(line, pad + 10, boxY + pad + i * lh);
  });
}

// 입력값 변경 시 캔버스 갱신
["inDate","inTime","inTag","inCat","inLoc","inNote"].forEach(id => {
  document.getElementById(id).addEventListener("input", () => {
    if (loadedImage) renderCanvas();
    if (id === "inCat") updateCatBar();
  });
  document.getElementById(id).addEventListener("change", () => {
    if (loadedImage) renderCanvas();
    if (id === "inCat") updateCatBar();
  });
});

function savePhoto() {
  if (!loadedImage) return;
  const canvas  = document.getElementById("canvas");
  const date    = document.getElementById("inDate").value.replace(/-/g,"");
  const time    = document.getElementById("inTime").value.replace(":","");
  const tag     = document.getElementById("inTag").value.trim().replace(/\s/g,"_") || "photo";
  const fname   = `SHAHEEN_${date}_${time}_${tag}.jpg`;
  const dataUrl = canvas.toDataURL("image/jpeg", 0.92);
  const a       = document.createElement("a");
  a.href        = dataUrl;
  a.download    = fname;
  document.body.appendChild(a);
  a.click();
  document.body.removeChild(a);
  showToast();
}

function showToast() {
  const t = document.getElementById("toast");
  t.style.display = "block";
  setTimeout(() => t.style.display = "none", 2500);
}

function resetAll() {
  loadedImage = null;
  document.getElementById("fileInput").value = "";
  document.getElementById("camBox").innerHTML = `
    <div class="cam-placeholder" id="placeholder">
      <div class="icon">📷</div>
      <p>탭하여 촬영 또는 갤러리 선택</p>
    </div>
    <div class="corner tl"></div><div class="corner tr"></div>
    <div class="corner bl"></div><div class="corner br"></div>`;
  document.getElementById("previewCard").style.display = "none";
  document.getElementById("inTag").value  = "";
  document.getElementById("inNote").value = "";
  const now = new Date();
  document.getElementById("inDate").value = now.toISOString().slice(0,10);
  document.getElementById("inTime").value = now.toTimeString().slice(0,5);
  window.scrollTo({top:0,behavior:"smooth"});
}
